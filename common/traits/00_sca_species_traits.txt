### Custom traits

trait_sca_ergivore = {
	sorting_priority = 20
	opposites = { "trait_plantoid_phototrophic" "trait_plantoid_radiotrophic" }
	initial = yes
	randomized = no
	
	species_potential_add = {
		always = no
	}
	species_possible_remove = {
		always = no
	}
	species_possible_merge_add = {
		always = no
	}

	allowed_archetypes = { BIOLOGICAL PRESAPIENT LITHOID MACHINE }
	opposites = { "trait_plantoid_budding" "trait_rooted" "trait_presapient_earthbound" }
	tags = { organic positive special }
	ai_weight = {
		weight = 0
	}
	# Limit characteristic stacking
	species_class = { SCA_ORGANIC }
	species_class_override = {
		hidden_trigger = {
			SCA_has_multiple_trophic_traits = no
		}
	}

	custom_tooltip_with_modifiers = TRAIT_VOIDLING_EFFECT
	resources = {
		category = planet_pops
		# Energy Upkeep
		upkeep = {
			trigger = {
				NOT = { has_trait = trait_voidling }
				is_being_purged = no
				is_sapient = yes
			}
			energy = @living_standard_energy_normal
		}
	}
}

	ai_weight = {
		weight = 0
	}

	custom_tooltip_with_modifiers = trait_sca_calamitous_birth_effects
	
	triggered_planet_growth_habitability_modifier = {
		potential = {
			exists = planet
			planet = {
