# SCA — Shadows of the Shroud trait overrides (keyed, load-last)
#
# This file REPLACES the old full-file copy of 17_shroud_species_traits.txt.
# Because it is named zz_* it loads after vanilla's 17_* file, so each trait
# defined here overrides vanilla by key while every trait NOT listed here
# (trait_blindness, trait_marked, and anything Paradox adds in future patches)
# comes straight from the live base game.
#
# Each block below = the CURRENT vanilla 4.4.x definition with ONLY the
# intended SCA deltas applied (marked with #SCA). Values that changed between
# 4.3.7 and 4.4.x that the old full-file copy was silently reverting
# (e.g. slave_cost trade vs energy, the trait_photoadaptive opposite) are left
# at their vanilla values on purpose.
########################################################################

	initial = no
	randomized = no

	species_possible_add = {
		always = no
	}
	species_possible_remove = {
		always = no
	}
	species_possible_merge_remove = {
		always = no
	}
	opposites = { "trait_weak" "trait_hollow_bones" "trait_photoadaptive" }

	custom_tooltip_with_modifiers = trait_tankbound_effects

	#SCA: drop the portrait lock, restrict by archetype instead (matches the

		pop_amenities_usage_mult = 0.5
		logistic_growth_mult = -0.25
	}
}
